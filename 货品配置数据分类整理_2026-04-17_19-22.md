# 解析 mrp_correlatives
def parse_mrp(val):
    if pd.isna(val) or val == '':
        return []
    try:
        items = json.loads(val)
        return [item.get('mrpBenefitId', '') for item in items]
    except:
        return []

# 解析 correlatives (关联商品)
def parse_corr(val):
    if pd.isna(val) or val == '':
        return ''
    try:
        items = json.loads(val)
        parts = []
        for item in items:
            pid = item.get('productId', '')
            sid = item.get('skuId', '')
            parts.append(f"productId={pid}, skuId={sid}")
        return '; '.join(parts)
    except:
        return ''

df['type_name'] = df['type'].map(type_map).fillna(df['type'])
df['mrp_list'] = df['mrp_correlatives'].apply(parse_mrp)
df['mrp_count'] = df['mrp_list'].apply(len)
